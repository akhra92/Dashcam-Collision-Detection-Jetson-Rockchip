#            ruamel.yaml scipy tqdm opencv-python fast-histogram onnxruntime
#        pip install "onnx==1.16.1"     # NEWER onnx removes onnx.mapping -> rknn load_onnx fails

#        pip install rknn_toolkit2-*-cp311-*x86_64.whl   # cp311 = match the env's Python

# 2) RK3588 BOARD — streaming inference (infer_rknn.py):

#      opencv-python
